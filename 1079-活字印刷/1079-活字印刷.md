### Java
``` java
class Solution {
    int[] count;
    public int numTilePossibilities(String tiles) {
        count = new int[26];
        for (char c : tiles.toCharArray()) {
            count[c - 'A']++;
        }
        return helper();
    }

    public int helper() {
        int res = 0;
        for (int i = 0; i < count.length; i++) {
            // 如果用完了，或者没有，直接跳过
            if (count[i] == 0) continue;
            // 代表我当前是一个分支
            res ++;
            // 做选择
            count[i]--;
            // 进入下一层，把下一层的有效分支+上
            res += helper();
            // 撤销选择
            count[i]++;
        }
        return res;
    }
}
```
